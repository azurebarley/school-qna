import streamlit as st
import google.generativeai as genai
import gspread
from oauth2client.service_account import ServiceAccountCredentials
import json, os

# 1. API 및 구글 시트 설정
genai.configure(api_key=st.secrets["GEMINI_API_KEY"])

def get_authorized_model():
    """안전하게 사용 가능한 모델명을 리스트에서 직접 추출합니다."""
    try:
        # 모델 리스트를 가져와서 'generateContent'가 가능한 모델 중 
        # 가장 안정적인 flash나 pro의 '전체 이름'을 가져옵니다.
        available_models = [m.name for m in genai.list_models() if 'generateContent' in m.supported_generation_methods]
        
        # 1. flash가 들어간 최신 모델을 먼저 찾고, 없으면 첫 번째 모델 선택
        target = next((m for m in available_models if "flash" in m), available_models[0])
        return target
    except:
        # 리스트 조회가 실패할 경우를 대비한 수동 경로
        return "models/gemini-1.5-flash-latest"

# 구글 시트 연결
scope = ["https://spreadsheets.google.com/feeds", "https://www.googleapis.com/auth/drive"]
creds_dict = json.loads(st.secrets["google_credentials"])
creds = ServiceAccountCredentials.from_json_keyfile_dict(creds_dict, scope)
client = gspread.authorize(creds)
sheet = client.open_by_url("https://docs.google.com/spreadsheets/d/1CDnTXib4J3C0QYrCKIMUcbZoS6Olez4Pkq3ltOJSH9U/edit").sheet1

# 2. 자료 불러오기
def load_data():
    if os.path.exists("매뉴얼.txt"):
        with open("매뉴얼.txt", "r", encoding="utf-8") as f:
            return f.read()
    return "등록된 지침 자료가 없습니다."

# 3. 화면 UI
st.set_page_config(page_title="시설개방 헬퍼", page_icon="🏫")
st.title("🏫 시설개방 스마트 질의응답")

with st.form("qna_form"):
    school = st.text_input("소속 학교명")
    question = st.text_area("궁금한 점을 상세히 적어주세요.")
    submit = st.form_submit_button("질문하기")

    if submit and question:
        if not question.strip():
            st.warning("질문을 입력해주세요.")
            st.stop()
            
        with st.spinner("지침서를 확인하며 답변을 생성 중입니다..."):
            try:
                # 404 에러 방지용: 시스템이 허용하는 모델명을 직접 가져옴
                model_full_name = get_authorized_model()
                model = genai.GenerativeModel(model_full_name)
                
                manual_content = load_data()
                prompt = f"당신은 안양과천교육지원청 AI입니다. 아래 지침을 근거로 답변하세요.\n\n[지침]\n{manual_content}\n\n[질문]: {question}"
                
                # 답변 생성
                response = model.generate_content(prompt)
                st.info(response.text)
                
                # 시트 기록
                sheet.append_row([school, question, response.text])
                st.success(f"✅ 완료 (사용 모델: {model_full_name})")
                
            except Exception as e:
                # 429 에러(사용량 초과) 발생 시 안내
                if "429" in str(e):
                    st.warning("⚠️ 현재 무료 버전의 일일 사용량 또는 분당 사용량이 초과되었습니다. 1분만 기다렸
